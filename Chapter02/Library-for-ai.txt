//  定义“书”这个数据类型
typedef struct {
    int id;                     //  书号（唯一标识）
    char name[50];              //  书名
    char author[20];            //  作者
    float price;                //  价格
} Book;

// 定义“顺序表”（即书架）
typedef struct {
    Book *data;                 //  指向动态数组的指针（存放书的地方）
    int length;                 //  当前书架上有多少本书
    int capacity;               //  书架最大能放多少本书
} SqList;  //  Sequence List
